package citadels;

import java.nio.charset.StandardCharsets;

import java.io.*;
import java.util.*;
import java.net.*;

public class App{
	
	private File cardsFile;
    private BufferedReader br;
    private boolean gameRuns = true;
    private boolean debugMode = false;
    private Map<String, Card> districtCards = new HashMap<>();
    private Game game;
    private static Scanner sc=new Scanner(System.in);
    private Random r = new Random();
    private int currentCharacterRank = 0;
    private List<String> cmd247 = new ArrayList<>(Arrays.asList("save", "load", "debug", "hand"));
    private boolean isLoaded = false;

	public App() throws UnsupportedEncodingException {
        BufferedReader br;

        try {
            //URL resource = getClass().getResource("/cards.tsv");
            InputStream inputStream = getClass().getResourceAsStream("/cards.tsv");
            
            if (inputStream == null) {
                throw new RuntimeException("cards.tsv not found!");
            }

            br = new BufferedReader(new InputStreamReader(inputStream, StandardCharsets.UTF_8));

            String line;
            br.readLine(); // skip header

            while ((line = br.readLine()) != null) {
                String[] vals = line.split("\t");
                String color = vals[2];
                String txt = vals.length == 5 ? vals[4] : "";
                int qty = Integer.parseInt(vals[1]);
                int cost = Integer.parseInt(vals[3]);

                districtCards.put(vals[0], new Card(vals[0], color, txt, qty, cost));
            }
        } catch (IOException e) {
            throw new RuntimeException("Failed to read cards.tsv", e);
        }
        this.game = new Game(districtCards);
	}

    
    public void runDebug(App app){

            System.out.println(Characters.getName(Characters.fromRank(i))+" : "+app.game.getPlayerCharMap().get(i).debug());
        }
        System.out.println("\nDiscarded Characters: "+app.game.getDiscardedCharacters().size());
        for(Characters c:app.game.getDiscardedCharacters()){
            System.out.println(Characters.getName(c));
        }
    }


    @SuppressWarnings("unused")

        String inp;
        Player p;
        Characters[] charList;
        System.out.println(app.currentCharacterRank+" is the current character rank." + this.game.getCurrentCharacterRank());

        System.out.println(app.isLoaded+" is the game loaded.");
        if(app.isLoaded){

        }
        System.out.println("Modded: "+charList.length);

            if(game.getDiscardedCharacters().contains(c)){

                inp = sc.next().trim();
                sc.nextLine();
                while(!inp.equalsIgnoreCase("t")){
                    if(cmd247.contains(inp.split(" ")[0])){

                        System.out.print("> ");
                        inp = sc.nextLine();
                    }
                    else{
                        System.out.print("It is not your turn yet. Please enter \'t\' to process turns.\n> ");
                        inp = sc.nextLine().trim();
                        //sc.nextLine();
                    }
                }
            }   
            else if(app.game.getPlayerCharMap().containsKey(c.getRank()) && app.game.getDeadCharacters().contains(c)){
                if(app.game.getPlayerCharMap().get(c.getRank()).getIsCharDead()){   
                    System.out.println("Player "+app.game.getPlayerCharMap().get(c.getRank()).getId()+" loses their turn because they were assassinated.");
                    inp = sc.nextLine();

                        if(cmd247.contains(inp.split(" ")[0])){
                            boolean temp = app.commands(app, inp.split(" "));
                            System.out.print("> ");
                            inp = sc.nextLine().trim();
                        }
                        else{
                            System.out.print("It is not your turn yet. Please enter \'t\' to process turns.\n> ");
                            inp = sc.nextLine().trim();
                        }
                    }
                }
            }
            else{
                p = app.game.getPlayerCharMap().get(c.getRank());
                System.out.println("Player "+p.getId()+" is the "+Characters.getName(c));
                if(p.getId()==1){
                    System.out.println("Your turn.");
                }    
                game.charAction(c, p.getId()!=1, p);

                //human code below
                if(p.getId()==1){
                    turn(false, p.getId());
                    boolean ret = true;
                    int count_build=0;
                    String cmd;
                    while(ret){
                        System.out.print("> ");
                        inp = sc.nextLine().trim();
                        cmd = (inp.indexOf(',')==-1?inp.split(" "):inp.split(","))[0];
                        if(cmd.equalsIgnoreCase("build") && inp.length()>5){

                        }
                        if(cmd.equalsIgnoreCase("action") && p.getCharCard().equalsIgnoreCase("Magician")){
                            System.out.println("You can use Magician's special ability only once per turn.");
                            continue;
                        }

                            continue;

                            ret = commands(app, inp.indexOf(',')==-1?inp.split(" "):inp.split(","));

                }

                    turn(true, p.getId());
                }

                inp = sc.nextLine().trim();
                while(!inp.equalsIgnoreCase("t")){
                    if(cmd247.contains(inp.split(" ")[0])){

                    }

                        System.out.print("It is not your turn yet. Please enter \'t\' to process turns.\n> ");
                        inp = sc.nextLine().trim();
                    }

    public void turn(boolean bot, int id){
        String inp;
        if(!bot){
            System.out.print("Collect 2 gold or draw two cards and pick one [gold/cards]:\n> ");
        }
        inp = bot?(this.game.getPlayers().get(id-1).getHand().size()>2?"gold":"cards"):sc.next();
        if(!bot){
            sc.nextLine();
        }
        while(!inp.equalsIgnoreCase("gold") && !inp.equalsIgnoreCase("cards")){
            System.out.print("Choose a valid option [gold/cards]\n> ");
            inp = sc.nextLine().trim();
        }
        if(!bot){
            System.out.println("You chose "+inp+".");
        }
        game.playerTurn(id, inp);
    }


    public boolean commands(App app, String[] cmd){
        switch (cmd[0]) {
                case "hand":
                    app.game.displayHand();
                    break;
                
                case "gold":
                    app.game.displayGold();
                    break;
                
                case "build":
                    if(cmd.length>1){
                        app.game.buildDistrict(cmd[1]);
                    }
                    else{
                        System.out.println("Specify the card index to build.");
                    }
                    break;

                case "citadel":
                case "city":
                case "list":
                    app.game.displayCity(cmd.length==1?"1":cmd[1]);
                    break;   
                
                case "action":
                    app.game.performAction(cmd, app.game.getPlayers().get(0));
                    break;

                case "info":
                    //two cases here one for purple building in hand and one for character
                    if(cmd.length>1){
                        app.game.showInfo(cmd[1]);
                    }
                    else{
                        System.out.println("Specify a character or card index.");
                    }
                    break;
                
                case "end":
                    app.game.endTurn();
                    break;

                case "all":
                    app.game.displayAll();
                    break;   
                
                case "save":
                    if(cmd.length>1){
                        System.out.println("Save|current character rank: "+app.currentCharacterRank);
                        app.game.saveGame(cmd[1], app.currentCharacterRank);
                        System.out.println("Do you wish to Continue the game. Enter your choice [yes/no]");
                        String inp = sc.nextLine();
                        while(!inp.equalsIgnoreCase("yes")){
                            if(inp.equalsIgnoreCase("no")){
                                System.exit(0);
                            }
                            System.out.print("Please choose a valid option [yes/no].\n> ");
                            inp = sc.nextLine();
                        }
                    }
                    else{
                        System.out.println("Please specify a filename to save game.");
                    }
                    break;

                case "load":
                    if(cmd.length>1){
                        app.game = app.game.loadGame(cmd[1], app.districtCards);
                        isLoaded = app.game!=null;
                        while(!isLoaded){
                            System.out.print("To retry enter (a) and to start a new game enter (b). Choose your option [a/b]\n> ");
                            String inp = sc.nextLine();
                            if(inp.equalsIgnoreCase("a")){
                                System.out.print("Enter filename,\n> ");
                                inp = sc.nextLine();
                                app.game = app.game.loadGame(inp, app.districtCards);
                                isLoaded = app.game!=null;
                            }
                            else if(inp.equalsIgnoreCase("b")){
                                break;
                            }
                            else{
                                System.out.println("Choose a valid option.");
                            }
                        }
                    }
                    else{
                        System.out.println("Please specify a filename to load game.");
                    }
                    break;
                
                case "debug":
                    //where npcs card will be visible on their turn
                    app.debugMode = !app.debugMode; //for turning it on or off when the command is entered
                    app.game.setDebugMode(app.debugMode);
                    System.out.println((app.debugMode?"Enabled":"Disabled")+" debug mode."+(app.debugMode?" You can now see all player's hands.":"You will no longer see all player\'s hands."));
                    break;
                
                case "help":
                default:
                    app.game.help();
                    break;
            }
            return !(cmd[0].equalsIgnoreCase("end"));
    }

    public static void main(String[] args) throws UnsupportedEncodingException {
        App app = new App();
        
        int playerCount = 0;
        System.out.print("Do you wish to load a previous game? [yes/no]\n> ");
        String inp = sc.nextLine().trim();
        if(inp.equalsIgnoreCase("yes")){
            System.out.print("Enter filename.\n> ");
            inp = sc.nextLine().trim();
            Game temp = app.game.loadGame(inp, app.districtCards); 
            app.isLoaded = temp!=null;
            if(temp!=null){
                app.game = temp;
                app.isLoaded = true;
                app.currentCharacterRank = app.game.getCurrentCharacterRank();
                // app.game.setGlobal(temp);
            }
            else{
                app.isLoaded = false;
            }
        }

        if(app.isLoaded==false){
            while(true){
                System.out.print("Enter how many players [4-7]:\n> ");
                playerCount = sc.nextInt();
                if(playerCount>3 && playerCount<8){
                    break;
                }
                else{
                    System.out.println("Invalid number try again.");
                }
            }            
            
            System.out.println("Shuffling deck...");
            System.out.println("Adding characters...");
            System.out.println("Dealing cards...");
            System.out.println("Starting Citadels with "+playerCount+" players...\nYou are player 1\n");
            app.game.begin(playerCount);
        }

        int roundCounter = 0;
        while(app.gameRuns){
            if(roundCounter>0){
                app.game.begin(playerCount);
                app.isLoaded = false; //resetting the loaded state for the next round
            }
            app.turnPhase(app);
            System.out.println("turn phase ended moving to process round.");
            app.gameRuns = !app.game.processesRound();
            app.game.setGameEnded(app.gameRuns);
            roundCounter++;
        }
    }
